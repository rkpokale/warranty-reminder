  <script>
    document.addEventListener("DOMContentLoaded", function () {
      const loginBtn = document.getElementById("loginBtn");
      loginBtn.addEventListener("click", function () {
        const user = document.getElementById("username").value.trim();
        const pass = document.getElementById("password").value.trim();
        if (user && pass) {
          document.getElementById("loginPage").style.display = "none";
          document.getElementById("shopPage").style.display = "flex";
          renderProducts();
        } else {
          alert("Please enter both username and password.");
        }
      });
    });

    const products = [
      { id: 1, name: "Jeans", image: "https://i.pinimg.com/736x/bc/4c/ec/bc4cecf5078e4bf4409cdb4e0321e9c7.jpg" },
      { id: 2, name: "Hoodie", image: "https://i.pinimg.com/1200x/3c/64/82/3c64821a423c2394816b81d528b4dad7.jpg" },
      { id: 3, name: "Sneaker", image: "https://i.pinimg.com/736x/2b/f0/df/2bf0dfcbd8d1ee4791d30ab117dde892.jpg" }
    ];

    let cart = [];

    function updateCartCount() {
      document.getElementById("cartCount").textContent = `Cart: ${cart.length}`;
    }

    function toggleCart(id) {
      const index = cart.indexOf(id);
      if (index === -1) {
        cart.push(id);
      } else {
        cart.splice(index, 1);
      }
      updateCartCount();
      renderProducts();
    }

    function renderProducts() {
      const grid = document.getElementById("productGrid");
      grid.innerHTML = "";
      products.forEach(p => {
        const inCart = cart.includes(p.id);
        grid.innerHTML += `
          <div class="product">
            <img src="${p.image}" alt="${p.name}">
            <h3>${p.name}</h3>
            <button onclick="toggleCart(${p.id})">
              ${inCart ? "Remove from Cart" : "Add to Cart"}
            </button>
          </div>
        `;
      });
    }

    function logout() {
      cart = [];
      updateCartCount();
      document.getElementById("shopPage").style.display = "none";
      document.getElementById("loginPage").style.display = "flex";
      document.getElementById("username").value = "";
      document.getElementById("password").value = "";
    }
  </script>
